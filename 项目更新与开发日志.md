**电压检测方案**

为实现路由器供电电压监测，针对 ESP32/ESP8266 模拟输入特性进行分压电路设计。

ESP32 与 ESP8266 模拟引脚耐压最大值为 3.3V，而路由器电源适配器输出为 12V，无法直接采集，因此采用电阻分压方案。

电路设计

选用 10kΩ 与 2kΩ 电阻串联构成分压电路；路由器电源适配器正极接 10kΩ 电阻一端；路由器电源适配器负极接 2kΩ 电阻一端，并与开发板 GND 共地；在 10kΩ 与 2kΩ 电阻连接点引出信号线，接入开发板模拟输入引脚（ADC）。

原理说明

通过电阻分压，将 12V 输入降至开发板可安全读取的电压范围，实现对路由器供电电压的实时检测。正常供电时，模拟输入引脚对地电压两伏，断电时，模拟输入引脚对地无电压。

const int AO\_DETECT = A0;       // 电压检测引脚（接分压电路，监测路由器12V供电）

const int RELAY\_PIN = D6;       // 继电器控制引脚（高电平吸合，接通酷态科供电）



// ====================== 核心参数配置 ======================

const int VOLT\_THRESHOLD = 400; // 电压触发阈值（A0正常2V对应模拟值≈620，400对应≈1.3V）

const int STABLE\_CHECK = 2;     // 防抖检测次数（2次，减少误触发）

// ====================== 初始化函数 ======================

void setup() {

&#x20; // 初始化串口通信（波特率9600，用于调试查看电压值和切换状态）

&#x20; Serial.begin(9600);

&#x20; // 等待串口初始化完成（避免开机前几秒串口无输出）

&#x20; delay(1000);

&#x20;

&#x20; pinMode(RELAY\_PIN, OUTPUT);

&#x20; 

&#x20; // 初始状态：继电器断开（低电平），避免开机误触发

&#x20; 

&#x20; // 串口打印初始化完成提示（方便调试）

&#x20; Serial.println("===== 路由器应急补电系统初始化完成 =====");

// ====================== 主循环函数（核心逻辑） ======================

void loop() {

&#x20; // 定义变量：记录连续检测到电压低于阈值的次数

&#x20; int lowVoltCount = 0;



&#x20; // --------------------- 第一步：多次防抖检测电压 ---------------------

&#x20; // 连续检测2次，每次间隔50ms，避免电压波动误判

&#x20; for (int i = 0; i < STABLE\_CHECK; i++) {

&#x20;   // 读取A0口模拟值（范围0-1023，对应0-3.3V）

&#x20;   int analogValue = analogRead(AO\_DETECT);

&#x20;   

&#x20;   // 串口打印实时检测值（方便校准阈值和调试）

&#x20;   Serial.print("第");

&#x20;   Serial.print(i + 1);

&#x20;   Serial.print("次检测 - A0模拟值:");

&#x20;   Serial.println(analogValue);

&#x20;   

&#x20;   // 若当前值低于阈值，计数+1

&#x20;   if (analogValue < VOLT\_THRESHOLD) {

&#x20;     lowVoltCount++;

&#x20;   }

&#x20;   

&#x20;   // 单次检测间隔（防抖核心，避免瞬间波动）

&#x20;   delay(STABLE\_INTERVAL);

&#x20; }



&#x20; // --------------------- 第二步：根据检测结果控制继电器 ---------------------

&#x20; // 连续2次检测到电压低于阈值 → 判定为市电断电，触发继电器

&#x20; if (lowVoltCount >= STABLE\_CHECK) {

&#x20;   digitalWrite(RELAY\_PIN, HIGH); // 高电平吸合继电器，接通酷态科供电

&#x20;   Serial.println(">>> 市电断电已接通应急供电 <<<");

&#x20; } 

&#x20; // 电压正常 → 继电器断开，切回路由器原装电源

&#x20; else {

&#x20;   digitalWrite(RELAY\_PIN, LOW);  // 低电平断开继电器，关闭酷态科供电

&#x20;   Serial.println(">>> 市电正常！已断开应急供电，切回原装电源 <<<");

&#x20; }



&#x20; // --------------------- 第三步：主检测间隔（控制整体响应速度） ---------------------

&#x20; Serial.println("----------------------------------------"); // 分隔线，方便看日志

&#x20; delay(CHECK\_INTERVAL);

}
